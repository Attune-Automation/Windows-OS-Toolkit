Step 1 - WSUU Install Brave

Step 1.1 - WSUU B Deploy Brave

Deploy the Brave browser setup file to the /software directory on the C drive.

Connect via RDP:

mstsc /admin /v:{targetserverwin}

Login as user {windowsadministrator} and open a command prompt.

For the latest release: https://github.com/brave/brave-browser/releases/latest

Copy the File(s) WIN_Util_Brave_64bit.unknown to the target node.

Extract and deploy the File(s) here:

/c$/software

Step 1.2 - WSUU B Install Brave

Install the Brave browser and create a Desktop shortcut for it.

Execute the following script: 

                
Set-Location c:\software

Start-Process .\BraveBrowserStandaloneSilentSetup.exe -Wait

$EXE = "C:\Windows\System32\config\systemprofile\AppData\Local\BraveSoftware\Brave-Browser\Application\brave.exe"
$DestPath = "C:\Users\Public\Desktop"

$DestLink = "$DestPath\Brave.lnk"
if (Test-Path "$DestLink") { Remove-Item "$DestLink" }

$WshShell = New-Object -comObject WScript.Shell
$Shortcut = $WshShell.CreateShortcut("$DestLink")
$Shortcut.TargetPath = "$EXE"
$Shortcut.Save()

Step 1.3 - WSUU B Remove Update Service

This will remove the Brave software update services.

Execute the following script: 

                
Unregister-ScheduledTask -TaskName "Brave*" -Confirm:$false

Step 4.2 - WSUU NP Install Notepad++

Install Notepad++, create a Desktop shortcut, associate .txt files to automatically open in Notepad++, and add 'send to' and 'open with' context menus.

Execute the following script: 

                
$ZIP="c:\software\npp.7.9.2.portable.zip"
$COMP_PLUGIN_ZIP="c:\software\ComparePlugin.v1.5.6.2.bin.zip"
$DEST="c:\Program Files\npp"
$EXE="$DEST\notepad++.exe"
$SEND_TO_TOYS_DIR="c:\Program Files\Send To Toys"

function Make-Shortcut ([string] $DestPath) {
    $DestLink = "$DestPath\Notepad ++.lnk"
    if (Test-Path "$DestLink") { Remove-Item "$DestLink" }
    
    $WshShell = New-Object -comObject WScript.Shell
    $Shortcut = $WshShell.CreateShortcut("$DestLink")
    $Shortcut.TargetPath = "$EXE"
    $Shortcut.Save()
}

if (Test-Path "$DEST") {
    "Removing old dir $DEST"
    Remove-Item "$DEST" -force -recurse
}

"Extracting $ZIP -> $DEST"
#& "C:\Program Files\7-Zip\7z.exe" x -o"$DEST" "$ZIP" 
Expand-Archive -LiteralPath "$ZIP" -DestinationPath "$DEST"

"Extracting compare plugin"
#& "C:\Program Files\7-Zip\7z.exe" x -o"$DEST\plugins" "$COMP_PLUGIN_ZIP"
Expand-Archive -LiteralPath "$COMP_PLUGIN_ZIP" -DestinationPath "$DEST\plugins"

"Creating start menu shortcut"
Make-Shortcut "C:\ProgramData\Microsoft\Windows\Start Menu"
Make-Shortcut "C:\Users\Public\Desktop"



"Associating txt files with Notepad++"
cmd /c @"
ftype txtfile="$EXE" %1 %*
"@
if (-Not ($?) -or "$LASTEXITCODE" -ne 0) {
    Write-Error "Failed to assicate text files with Notepad++"
}

"Adding notepad++ to 'Send To' Menu"
push-location $SEND_TO_TOYS_DIR
.\SendToAdd.exe "$EXE"

pop-location

"Adding notepad++ to 'Open With' Menu"
$escapedExe = $EXE -replace "\\", "\\\\";

$reg = @"
Windows Registry Editor Version 5.00

[HKEY_CLASSES_ROOT\*\shell\Open with Notepad++]

[HKEY_CLASSES_ROOT\*\shell\Open with Notepad++\command]
@="\"$escapedExe\" \"%1\""
"Icon"="($escapedExe),0"

"@

$tmp = New-TemporaryFile
out-file -filePath $tmp.FullName -InputObject $reg
reg import $tmp.FullName /reg:64
Remove-Item $tmp.FullName

Remove-Item $ZIP
Remove-Item $COMP_PLUGIN_ZIP

"Fixing Load langs.xml Failed Error in Notepad++"
# Rename-Item -Path "$($DEST)\langs.xml" -NewName "langs.xml.bad"
Copy-Item "$($DEST)\langs.model.xml" -Destination  "$($DEST)\langs.xml"

"Fixing stylers.xml popup in Notepad++"
Copy-Item "$($DEST)\stylers.model.xml" -Destination "$($DEST)\stylers.xml"

$REG_PATH = 'Registry::HKEY_LOCAL_MACHINE\System\CurrentControlSet\Control\Session Manager\Environment'
$path = (Get-ItemProperty -Path "$REG_PATH" -Name PATH).Path

if ($path -Match 'WinSCP') {
    "WinSCP already added to path"
} else {
    "Adding WinSCP to the PATH"
    $path = $path + ";C:\Program Files (x86)\WinSCP"
    Set-ItemProperty -Path "$REG_PATH" -Name PATH -Value $path
}
